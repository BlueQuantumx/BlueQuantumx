inline ll qpow(ll base, ll exp, const int &mod) {
  ll res = 1;
  while (exp) {
    if (exp & 1) res = res * base % mod;
    base = base * base % mod;
    exp >>= 1;
  }
  return res;
}
​
void exgcd(ll a, ll b, ll &x, ll &y) {
  if (b == 0) return x = 1, y = 0, void();
  exgcd(b, a % b, y, x);
  y -= a / b * x;
}
